## 1. 시스템 개요 및 아키텍처

### 1.1 개요

AI 감정 반응 엔진(Emotional Response Engine, ERE)은 사용자의 감정 상태를 실시간으로 감지하고, 베니 캐릭터의 표정/행동과 정원 환경에 반영하는 핵심 서브시스템입니다.

**주요 기능:**
- 사용자 입력(텍스트, UI 인터랙션, 체크인 데이터) 기반 감정 분석
- 5감정 상태(기쁨/평온/피곤/슬픔/불안) 실시간 추론
- VARCO AI와 연동하여 베니의 자연어 응답 생성
- 감정 결과를 Unity 환경(날씨, 오브젝트)에 동기화
- 감정 이력 로그 저장 및 패턴 분석

#### 2.2.3 다중 입력 통합 (Weighted Fusion)

```python
def fuse_emotion_signals(signals: list[WeightedSignal]) -> FinalEmotionState:
    """
    여러 입력 소스의 감정 신호를 가중 평균으로 통합
    """
    emotion_scores = {
        "joy": 0.0, "calm": 0.0,
        "tired": 0.0, "sad": 0.0, "anxious": 0.0
    }
    
    total_weight = 0
    for signal in signals:
        for emotion in emotion_scores:
            emotion_scores[emotion] += signal.vector[emotion] * signal.weight
        total_weight += signal.weight
    
    # 정규화
    for emotion in emotion_scores:
        emotion_scores[emotion] /= total_weight
    
    # 최고 점수 감정 선택
    primary = max(emotion_scores, key=emotion_scores.get)
    intensity = score_to_intensity(emotion_scores[primary])  # 0.0~1.0 → 1~5
    
    return FinalEmotionState(
        primary=primary,
        scores=emotion_scores,
        intensity=intensity,
        confidence=emotion_scores[primary]
    )
```

#### 2.2.4 강도 레벨 변환

```python
def score_to_intensity(score: float) -> int:
    """
    확률 점수(0.0~1.0)를 1~5 강도 레벨로 변환
    """
    thresholds = [0.2, 0.4, 0.6, 0.8]
    for i, threshold in enumerate(thresholds):
        if score < threshold:
            return i + 1
    return 5
```

```python
EMOTION_ENV_MAP = {
    "joy": {
        1: EnvironmentParams(weather="sunny", sky="#87CEEB", plant="budding",
                             particle="butterflies_few", sound="birds_calm", brightness=0.8),
        3: EnvironmentParams(weather="sunny", sky="#4FC3F7", plant="blooming",

        5: EnvironmentParams(weather="glowing", sky="#FFD700", plant="full_bloom_glow",

    "sad": {
        1: EnvironmentParams(weather="drizzle", sky="#90A4AE", plant="rain_drops",
                             particle="rain_light", sound="rain_soft", brightness=0.5),
        3: EnvironmentParams(weather="rain", sky="#546E7A", plant="drooping",
                             particle="rain_medium", sound="rain_medium", brightness=0.35),
        5: EnvironmentParams(weather="heavy_rain", sky="#37474F", plant="severely_drooping",
                             particle="rain_heavy", sound="thunder_distant", brightness=0.2),
    },
    # ... 기타 감정 정의

def get_environment_params(emotion: str, intensity: int) -> EnvironmentParams:
    """
    감정과 강도를 받아 가장 가까운 환경 파라미터 반환
    강도는 1, 3, 5 기준점에서 선형 보간
    """
    emotion_map = EMOTION_ENV_MAP.get(emotion, EMOTION_ENV_MAP["calm"])
    
    # 강도 기준점 찾기 및 보간
    breakpoints = sorted(emotion_map.keys())
    lower = max(b for b in breakpoints if b <= intensity)
    upper = min(b for b in breakpoints if b >= intensity)
    
    if lower == upper:
        return emotion_map[lower]
    
    t = (intensity - lower) / (upper - lower)
    return interpolate_env(emotion_map[lower], emotion_map[upper], t)
```

#### 4.2.1 Redis 캐시 레이어

```python
# 캐시 키 설계
class CacheKeyBuilder:
    
    @staticmethod
    def emotion_response(emotion: str, intensity: int, context_hash: str) -> str:
        """유사한 감정 상태에 대한 응답 캐싱"""
        return f"ere:response:{emotion}:{intensity}:{context_hash}"
    
    @staticmethod
    def user_emotion_state(user_id: str) -> str:
        """사용자의 현재 감정 상태 캐싱 (TTL: 30분)"""
        return f"ere:state:{user_id}:current"
    
    @staticmethod
    def env_params(emotion: str, intensity: int) -> str:
        """환경 파라미터 캐싱 (정적 - TTL 없음)"""
        return f"ere:env:{emotion}:{intensity}"

# TTL 설정
CACHE_TTL = {
```

```python
# 베니 응답 스트리밍 처리
async def stream_benny_response(ws: WebSocket, emotion_state: EmotionState):
    """
    VARCO 응답을 스트리밍으로 Unity 클라이언트에 전달
```

### 5.4 에러 처리

```python
class VARCOErrorHandler:
    
    RETRY_CODES = [429, 500, 502, 503, 504]
    
    async def call_with_retry(self, request: VARCORequest, max_retries: int = 3):
        """
        지수 백오프 재시도 + 폴백 응답
        """
        for attempt in range(max_retries):
            try:
                return await varco_client.call(request)
            
            except RateLimitError as e:
                # 429: 큐에 넣고 대기
                wait_time = e.retry_after or (2 ** attempt)
                await asyncio.sleep(wait_time)
            
            except ServerError as e:
                if attempt < max_retries - 1:
                    await asyncio.sleep(2 ** attempt)
                else:
                    return self.get_fallback_response(request.emotion_context)
            
            except TimeoutError:
                return self.get_cached_fallback(request.emotion_context)
        
        return self.get_fallback_response(request.emotion_context)
    
    def get_fallback_response(self, emotion_context: dict) -> str:
        """
        VARCO API 장애 시 로컬 폴백 응답 (사전 준비된 스크립트)
        """
        emotion = emotion_context.get("primary", "calm")
        return FALLBACK_RESPONSES[emotion][
            emotion_context.get("intensity", 3)
        ]
```

```python
class EmotionPatternAnalyzer:
    
    def analyze_weekly_pattern(self, user_id: str, weeks: int = 4) -> WeeklyPattern:
        """
        최근 N주 감정 패턴 분석
        """
        data = self.fetch_emotion_series(user_id, days=weeks*7)
        
        return WeeklyPattern(
            dominant_emotion=self._get_dominant(data),
            trend=self._calculate_trend(data),        # improving|stable|declining
            peak_times=self._find_peak_times(data),   # 감정이 강한 시간대
            weekly_rhythm=self._analyze_weekly(data),  # 요일별 패턴
            consecutive_negative_days=self._count_consecutive_negative(data)
        )
    
    def _calculate_trend(self, data: list) -> str:
        """
        최근 7일 vs 이전 7일 감정 점수 비교
        긍정 감정(joy+calm) 증감으로 트렌드 판단
        """
        recent = data[-7:]
        previous = data[-14:-7]
        
        recent_positive = np.mean([d.joy + d.calm for d in recent])
        prev_positive = np.mean([d.joy + d.calm for d in previous])
        
        delta = recent_positive - prev_positive
        if delta > 0.1:   return "improving"
        if delta < -0.1:  return "declining"
        return "stable"
    
    def check_crisis_indicators(self, user_id: str) -> CrisisIndicators:
        """
        위기 조기 감지 지표 확인
        """
        recent = self.fetch_emotion_series(user_id, days=3)
        
        return CrisisIndicators(
            consecutive_high_sad=self._count_consecutive(recent, "sad", min_intensity=4),
            consecutive_high_anxious=self._count_consecutive(recent, "anxious", min_intensity=4),
            total_app_absence_days=self._count_absence(user_id, days=7),
            rapid_decline=self._detect_rapid_decline(recent)
        )
```
