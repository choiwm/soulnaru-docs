 **추가 구현:** FirestoreManager.PreloadEmotionLogs() 메서드 신규 추가 
 // FirestoreManager.cs — 신규 메서드
public async Task Dictionary string, List EmotionLog PreloadEmotionLogs(
 string userId, int days = 30)
{
 var result = new Dictionary string, List EmotionLog ();
 var since = DateTime.UtcNow.AddDays(-days);
 var query = db.Collection("EmotionLog")
 .WhereEqualTo("userId", userId)
 .WhereGreaterThanOrEqualTo("createdAt", since)
 .OrderByDescending("createdAt");
 var snapshot = await query.GetSnapshotAsync();
 foreach (var doc in snapshot.Documents) {
 var log = doc.ConvertTo EmotionLog ();
 var key = log.CreatedAt.ToLocalTime().ToString("yyyy-MM-dd");
 if (!result.ContainsKey(key)) result[key] = new List EmotionLog ();
 result[key].Add(log);
 }
 return result;
}

 Day 13 3분 타이머 — MissionTimer 컴포넌트 신규 설계 
 **배경:** Day 13 "불편함과 잠시 앉기" 미션에 3분 타이머 필요 — 현재 MissionManager에 타이머 기능 없음 
 // MissionTimer.cs — 신규 컴포넌트
public class MissionTimer : MonoBehaviour {
 [SerializeField] private float durationSeconds = 180f; // 3분
 [SerializeField] private TextMeshProUGUI timerText;
 [SerializeField] private Slider progressSlider;

 public UnityEvent onTimerComplete;
 public UnityEvent onTimerCancelled;

 private float _elapsed = 0f;
 private bool _isRunning = false;

 public void StartTimer(float seconds = -1) {
 if (seconds > 0) durationSeconds = seconds;
 _elapsed = 0f;
 _isRunning = true;
 StartCoroutine(TimerCoroutine());
 }

 public void CancelTimer() {
 _isRunning = false;
 onTimerCancelled?.Invoke();
 }

 private IEnumerator TimerCoroutine() {
 while (_elapsed durationSeconds _isRunning) {
 _elapsed += Time.deltaTime;
 float remaining = durationSeconds - _elapsed;
 // UI 업데이트
 timerText.text = $"{(int)(remaining/60):D2}:{(int)(remaining%60):D2}";
 progressSlider.value = _elapsed / durationSeconds;
 yield return null;
 }
 if (_isRunning) onTimerComplete?.Invoke();
 }
}
